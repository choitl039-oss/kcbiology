    <script>
        const canvas = document.getElementById('leafCanvas');
        const ctx = canvas.getContext('2d');

        let particles = [];
        let activeStep = 0; 

        // Spongy Mesophyll Cells setup
        const cells = [
            { x: 280, y: 220, r: 45 }, { x: 440, y: 190, r: 50 },
            { x: 600, y: 240, r: 40 }, { x: 330, y: 350, r: 45 },
            { x: 500, y: 380, r: 55 }, { x: 660, y: 350, r: 45 }
        ];

        // Stoma Setup
        const stoma = { x: 500, y: 520, width: 65 };

        // Text Content for Explanations
        const stepContent = {
            0: {
                title: "Overall Process",
                text: "Observe the entire continuous cycle from water entering the xylem to water vapour exiting the stomata.",
                note: "",
                color: "#8b5cf6"
            },
            1: {
                title: "Step 1",
                text: "Water is pulled up the stem and enters the leaf through the <strong>xylem vessels</strong>. It then moves into the surrounding spongy mesophyll cells via osmosis.",
                note: "",
                color: "#3b82f6"
            },
            2: {
                title: "Step 2",
                text: "Water diffuses out of the mesophyll cells to form a (5) <strong>water film on the cell surface.</strong>",
                note: "(The water is moving from inside the cell to the edge based on a concentration gradient)",
                color: "#3b82f6"
            },
            3: {
                title: "Step 3",
                text: "Water of the water film (6) <strong>evaporates to form water vapour. The water vapour moves into the air space among the mesophyll cells.</strong>",
                note: "",
                color: "#3b82f6"
            },
            4: {
                title: "Step 4",
                text: "Water vapour in the air space (7) <strong>diffuses through the stomata to the atmosphere.</strong>",
                note: "",
                color: "#3b82f6"
            }
        };

        class WaterParticle {
            constructor() {
                this.reset();
                if (activeStep === 0 || activeStep === 1) this.y = 550 + Math.random() * 300; 
            }

            reset() {
                this.targetCell = cells[Math.floor(Math.random() * cells.length)];
                this.timer = 0;
                this.size = 4 + Math.random() * 2;
                this.color = 'rgba(41, 128, 185, 1)'; 
                this.speedMult = 1.2 + Math.random() * 0.6;

                if (activeStep === 0 || activeStep === 1) {
                    this.state = 1; // Start in Xylem
                    this.x = 50 + Math.random() * 40; 
                    this.y = 550 + Math.random() * 150; 
                } 
                else if (activeStep === 2) {
                    this.state = 2; // Start inside Cell
                    this.x = this.targetCell.x + (Math.random() - 0.5) * 30;
                    this.y = this.targetCell.y + (Math.random() - 0.5) * 30;
                }
                else if (activeStep === 3) {
                    this.state = 3; // Start as Water Film
                    this.angleOnCell = Math.random() * Math.PI * 2;
                    this.x = this.targetCell.x + (this.targetCell.r + 3) * Math.cos(this.angleOnCell);
                    this.y = this.targetCell.y + (this.targetCell.r + 3) * Math.sin(this.angleOnCell);
                }
                else if (activeStep === 4) {
                    this.state = 4; // Start in Air Space
                    this.size += 2; 
                    this.color = 'rgba(133, 193, 233, 0.8)'; 
                    this.x = 250 + Math.random() * 450;
                    this.y = 150 + Math.random() * 300;
                }
            }

            update() {
                if (this.state === 1) {
                    if (this.y > this.targetCell.y) {
                        this.y -= 2 * this.speedMult; 
                        this.x += (Math.random() - 0.5); 
                    } else {
                        this.x += 2 * this.speedMult;
                        const dx = this.targetCell.x - this.x;
                        const dy = this.targetCell.y - this.y;
                        if (Math.hypot(dx, dy) <= this.targetCell.r - 8) this.state = 2;
                    }
                } 
                else if (this.state === 2) {
                    this.timer++;
                    this.x += (Math.random() - 0.5) * 2;
                    this.y += (Math.random() - 0.5) * 2;
                    
                    const dist = Math.hypot(this.x - this.targetCell.x, this.y - this.targetCell.y);
                    if (dist > this.targetCell.r - 10) {
                        this.x = this.targetCell.x; 
                        this.y = this.targetCell.y;
                    }

                    if (activeStep === 0 && this.timer > 30) {
                        this.state = 3; 
                        this.angleOnCell = Math.atan2(this.y - this.targetCell.y, this.x - this.targetCell.x);
                        this.timer = 0;
                    } else if (activeStep === 1 && this.timer > 120) {
                        this.reset();
                    } else if (activeStep === 2 && this.timer > 40) {
                        this.state = 3; 
                        this.angleOnCell = Math.atan2(this.y - this.targetCell.y, this.x - this.targetCell.x);
                        this.timer = 0;
                    }
                }
                else if (this.state === 3) {
                    this.timer++;
                    this.angleOnCell += 0.015 * this.speedMult;
                    this.x = this.targetCell.x + (this.targetCell.r + 3) * Math.cos(this.angleOnCell);
                    this.y = this.targetCell.y + (this.targetCell.r + 3) * Math.sin(this.angleOnCell);
                    
                    if (activeStep === 0 && this.timer > 60) {
                        this.state = 4; 
                        this.size += 2; 
                        this.color = 'rgba(133, 193, 233, 0.8)'; 
                        this.timer = 0;
                    } else if (activeStep === 2 && this.timer > 150) {
                        this.reset();
                    } else if (activeStep === 3 && this.timer > 50) {
                        this.state = 4; 
                        this.size += 2; 
                        this.color = 'rgba(133, 193, 233, 0.8)'; 
                        this.timer = 0;
                    }
                }
                else if (this.state === 4) {
                    this.timer++;
                    this.x += (Math.random() - 0.5) * 6 * this.speedMult; 
                    this.y += (Math.random() - 0.5) * 6 * this.speedMult;

                    const dx = stoma.x - this.x;
                    const dy = stoma.y - this.y;
                    this.x += (dx * 0.005) * this.speedMult;
                    this.y += (dy * 0.01) * this.speedMult;

                    if ((activeStep === 0 || activeStep === 4) && this.y > 480 && Math.abs(this.x - stoma.x) < 80) {
                        this.state = 5;
                    } else if (activeStep === 3) {
                        if (this.y > 490) this.y = 490; 
                        if (this.timer > 200) this.reset();
                    }
                }
                else if (this.state === 5) {
                    this.y += 3 * this.speedMult;
                    this.x += (Math.random() - 0.5) * 5;
                    if (this.y > 600) this.reset();
                }
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fillStyle = this.color;
                ctx.fill();
            }
        }

        // Initialize particles
        for (let i = 0; i < 150; i++) {
            particles.push(new WaterParticle());
        }

        function drawLabel(text, x, y, lineToX, lineToY) {
            ctx.fillStyle = 'rgba(255, 255, 255, 0.9)';
            ctx.fillRect(x - 5, y - 16, ctx.measureText(text).width + 10, 24);
            ctx.fillStyle = '#0f172a';
            ctx.font = '15px bold Segoe UI';
            ctx.fillText(text, x, y);

            if (lineToX && lineToY) {
                ctx.strokeStyle = '#64748b';
                ctx.lineWidth = 1.5;
                ctx.beginPath();
                ctx.moveTo(x - 5, y - 4);
                ctx.lineTo(lineToX, lineToY);
                ctx.stroke();
            }
        }

        function drawLeafStructure() {
            // Xylem
            ctx.fillStyle = '#bdc3c7'; 
            ctx.fillRect(20, 0, 100, 550);
            ctx.strokeStyle = '#95a5a6';
            ctx.lineWidth = 2;
            for(let i=30; i<=110; i+=20) {
                ctx.beginPath();
                ctx.moveTo(i, 0);
                ctx.lineTo(i, 550);
                ctx.stroke();
            }
            
            // Upper Epidermis 
            ctx.fillStyle = '#a9dfbf';
            ctx.fillRect(120, 0, 680, 45);
            
            // Palisade Mesophyll
            ctx.fillStyle = '#27ae60';
            for(let i=130; i<800; i+=40) {
                ctx.fillRect(i, 50, 30, 90); 
                ctx.strokeStyle = '#1e8449';
                ctx.strokeRect(i, 50, 30, 90);
            }

            // Lower Epidermis
            ctx.fillStyle = '#a9dfbf';
            ctx.fillRect(120, 500, stoma.x - 120 - stoma.width/2, 50); 
            ctx.fillRect(stoma.x + stoma.width/2, 500, 800 - stoma.x, 50); 
            
            // Guard Cells
            ctx.fillStyle = '#2ecc71';
            let stomaGap = (activeStep === 0 || activeStep === 4) ? 28 : 0; 
            
            ctx.beginPath();
            ctx.ellipse(stoma.x - stomaGap/2 - 16, 525, 22, 14, 0, 0, Math.PI*2);
            ctx.fill();
            ctx.stroke();
            
            ctx.beginPath();
            ctx.ellipse(stoma.x + stomaGap/2 + 16, 525, 22, 14, 0, 0, Math.PI*2);
            ctx.fill();
            ctx.stroke();

            // Spongy Mesophyll Cells
            cells.forEach(cell => {
                if (activeStep === 0 || activeStep >= 2) {
                    ctx.beginPath();
                    ctx.arc(cell.x, cell.y, cell.r + 7, 0, Math.PI * 2);
                    ctx.fillStyle = 'rgba(133, 193, 233, 0.4)';
                    ctx.fill();
                }

                ctx.beginPath();
                ctx.arc(cell.x, cell.y, cell.r, 0, Math.PI * 2);
                ctx.fillStyle = '#2ecc71'; 
                ctx.fill();
                ctx.strokeStyle = '#27ae60';
                ctx.lineWidth = 3;
                ctx.stroke();
            });

            ctx.font = '15px bold Segoe UI';
            drawLabel('Upper Epidermis', 640, 30);
            drawLabel('Palisade Mesophyll', 640, 105);
            drawLabel('Air Space', 680, 260);
            drawLabel('Spongy Mesophyll', 640, 420, 615, 390);
            drawLabel('Lower Epidermis', 640, 530);
            drawLabel('Xylem Vessels', 12, 40);
            drawLabel('Guard Cell / Stoma', 580, 535, 540, 528);
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            drawLeafStructure();
            particles.forEach(p => { p.update(); p.draw(); });
            requestAnimationFrame(animate);
        }

        // Logic to switch steps and UI text
        window.setStep = function(stepNum) {
            activeStep = stepNum;
            
            // Update UI Buttons
            document.querySelectorAll('.step-btn').forEach(btn => btn.classList.remove('active'));
            document.getElementById('btn-' + stepNum).classList.add('active');

            // Update UI Text Panel
            const info = stepContent[stepNum];
            const explanationBox = document.getElementById('explanation-box');
            
            let htmlContent = `<h3>${info.title}</h3><p>${info.text}</p>`;
            if(info.note) htmlContent += `<span class="note">${info.note}</span>`;
            
            explanationBox.innerHTML = htmlContent;
            explanationBox.style.borderLeftColor = info.color;

            // Reset particles
            particles.forEach(p => p.reset());
        }

        // Initialize first view
        setStep(0);
        animate();
    </script>
