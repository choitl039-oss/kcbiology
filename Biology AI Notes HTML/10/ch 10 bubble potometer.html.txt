    <script>
        /* =========================================
           PART 1: MATHEMATICAL MODELS & GRAPHING
        ========================================= */
        const factorDescriptions = {
            light: "<strong>Light Intensity:</strong> As light increases, stomata open wider to allow more CO₂ in for photosynthesis. This proportionately increases the rate of transpiration until the stomata are fully open, causing the rate to plateau.",
            temp: "<strong>Temperature:</strong> As temperature increases, the kinetic energy of water molecules increases. This accelerates the rate of evaporation from the mesophyll surfaces into the air spaces, exponentially increasing transpiration.",
            wind: "<strong>Air Movement (Wind Speed):</strong> As wind speed increases, it blows away the humid diffusion shell around the leaf, steepening the concentration gradient and increasing transpiration. <strong>However, when wind speed further increases to extreme levels</strong>, excessive water loss causes guard cells to lose turgor, closing the stomata and causing a sharp drop in the transpiration rate.",
            humidity: "<strong>Relative Humidity:</strong> As humidity increases, the water vapor concentration in the atmosphere rises. This reduces the concentration gradient between the inside of the leaf and the outside air, causing the rate of transpiration to steadily decrease."
        };

        const mathModels = {
            light: (x) => { return 100 * (1 - Math.exp(-0.05 * x)); }, 
            temp: (x) => { return Math.pow(x, 2) / 20.25; }, 
            wind: (x) => { 
                if(x < 50) return 20 + (80 * Math.sin((x / 50) * (Math.PI / 2)));
                else return 100 * Math.exp(-0.08 * (x - 50)); 
            },
            humidity: (x) => { return 110 - x; } 
        };

        let activeFactor = 'light';
        const ctxChart = document.getElementById('factorChart').getContext('2d');
        let factorChart;

        function updateChart() {
            const currentVal = parseFloat(document.getElementById(`slide-${activeFactor}`).value);
            
            let curveData = [];
            let maxDomain = (activeFactor === 'temp') ? 45 : 100;
            let minDomain = (activeFactor === 'temp') ? 10 : 0;
            if(activeFactor === 'humidity') minDomain = 10;

            for(let i = minDomain; i <= maxDomain; i += (maxDomain-minDomain)/50) {
                curveData.push({ x: i, y: mathModels[activeFactor](i) });
            }

            let currentPoint = [{ x: currentVal, y: mathModels[activeFactor](currentVal) }];

            let xTitle = activeFactor.charAt(0).toUpperCase() + activeFactor.slice(1);
            if(activeFactor === 'temp') xTitle = 'Temperature (°C)';
            if(activeFactor === 'wind') xTitle = 'Wind Speed (km/h)';
            if(activeFactor === 'humidity') xTitle = 'Relative Humidity (%)';
            if(activeFactor === 'light') xTitle = 'Light Intensity (%)';

            if(factorChart) factorChart.destroy();

            factorChart = new Chart(ctxChart, {
                type: 'scatter',
                data: {
                    datasets: [
                        {
                            type: 'scatter',
                            label: 'Current Condition',
                            data: currentPoint,
                            backgroundColor: '#ef4444',
                            pointRadius: 8,
                            pointHoverRadius: 10,
                            pointBorderColor: 'white',
                            pointBorderWidth: 2,
                            z: 10
                        },
                        {
                            type: 'line',
                            label: 'Theoretical Rate',
                            data: curveData,
                            borderColor: '#3b82f6',
                            backgroundColor: 'rgba(59, 130, 246, 0.1)',
                            borderWidth: 3,
                            fill: true,
                            pointRadius: 0,
                            tension: 0.4
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: { legend: { position: 'top' } },
                    scales: {
                        x: { 
                            title: { display: true, text: xTitle, font: { weight: 'bold' } },
                            min: minDomain, max: maxDomain
                        },
                        y: { 
                            title: { display: true, text: 'Transpiration Rate', font: { weight: 'bold' } },
                            min: 0, max: 110,
                            ticks: { display: false } 
                        }
                    }
                }
            });

            document.getElementById('graph-text').innerHTML = factorDescriptions[activeFactor];
        }

        window.selectFactor = function(factor) {
            activeFactor = factor;
            document.querySelectorAll('.slider-group').forEach(el => el.classList.remove('active'));
            document.getElementById(`group-${factor}`).classList.add('active');
            updateChart();
        };

        /* =========================================
           PART 2: POTOMETER ANIMATION & LOGIC
        ========================================= */
        const potoCanvas = document.getElementById('potoCanvas');
        const potoCtx = potoCanvas.getContext('2d');
        const playBtn = document.getElementById('playBtn');
        const resetBtn = document.getElementById('resetBtn');

        let isPlaying = true;
        let isResetting = false;
        let totalUptake = 0; 
        
        // Coordinates for a 600px width canvas
        const startBubbleX = 510;
        const endBubbleX = 200;
        let bubbleX = startBubbleX;

        let vaporParticles = [];
        let usedParticles = [];
        const leaves = [ {x:70, y:120}, {x:140, y:90}, {x:50, y:60}, {x:120, y:40} ];

        class Particle {
            constructor(x, y, type) {
                this.x = x + (Math.random() - 0.5) * 30;
                this.y = y + (Math.random() - 0.5) * 20;
                this.life = 0;
                this.maxLife = type === 'vapor' ? 40 + Math.random()*20 : 60;
                this.type = type;
                this.size = 2 + Math.random() * 2;
            }
            update(rateModifier) {
                this.life++;
                if(this.type === 'vapor') {
                    this.y -= 2 * Math.max(0.5, rateModifier); 
                    this.x += (Math.random() - 0.5) * 4; // Wind drift effect can be added here
                }
            }
            draw() {
                const alpha = Math.max(0, 1 - (this.life / this.maxLife));
                potoCtx.globalAlpha = alpha;
                if(this.type === 'vapor') {
                    potoCtx.beginPath(); potoCtx.arc(this.x, this.y, this.size, 0, Math.PI*2);
                    potoCtx.fillStyle = '#bae6fd'; potoCtx.fill();
                } else {
                    potoCtx.fillStyle = '#16a34a'; potoCtx.font = 'bold 16px Arial';
                    potoCtx.fillText('+', this.x, this.y);
                }
                potoCtx.globalAlpha = 1.0;
            }
        }

        function drawPotometer() {
            potoCtx.clearRect(0, 0, potoCanvas.width, potoCanvas.height);

            // Desk line
            potoCtx.strokeStyle = '#cbd5e1'; potoCtx.lineWidth = 4;
            potoCtx.beginPath(); potoCtx.moveTo(20, 320); potoCtx.lineTo(580, 320); potoCtx.stroke();

            // Water (Blue fill)
            potoCtx.fillStyle = 'rgba(59, 130, 246, 0.2)';
            potoCtx.fillRect(50, 150, 40, 120); // Plant tube
            potoCtx.fillRect(50, 250, 500, 20); // Capillary
            potoCtx.fillRect(140, 80, 40, 170); // Reservoir tube
            potoCtx.beginPath(); // Reservoir Funnel water
            potoCtx.moveTo(140, 80); potoCtx.lineTo(120, 40); potoCtx.lineTo(200, 40); potoCtx.lineTo(180, 80); potoCtx.fill();

            // Beaker
            potoCtx.fillRect(520, 200, 60, 120); 

            // Air Bubble
            potoCtx.fillStyle = '#ffffff'; potoCtx.strokeStyle = '#ef4444'; potoCtx.lineWidth = 2;
            potoCtx.beginPath(); potoCtx.roundRect(bubbleX, 252, 20, 16, 8);
            potoCtx.fill(); potoCtx.stroke();
            potoCtx.fillStyle = '#ef4444'; potoCtx.font = 'bold 11px Arial';
            potoCtx.fillText('Air Bubble', bubbleX - 15, 245);

            // Ruler
            potoCtx.fillStyle = '#fef08a'; potoCtx.fillRect(190, 275, 330, 20);
            potoCtx.strokeStyle = '#000'; potoCtx.lineWidth = 1; potoCtx.strokeRect(190, 275, 330, 20);
            potoCtx.fillStyle = '#000'; potoCtx.font = '10px Arial';
            for(let i=0; i<=10; i++) {
                let lx = 510 - (i * 31);
                potoCtx.beginPath(); potoCtx.moveTo(lx, 275); potoCtx.lineTo(lx, 282); potoCtx.stroke();
            }

            // Draw Glass Apparatus
            potoCtx.strokeStyle = 'rgba(148, 163, 184, 0.8)'; potoCtx.lineWidth = 6; potoCtx.lineJoin = 'round';
            potoCtx.beginPath();
            // Plant tube
            potoCtx.moveTo(50, 120); potoCtx.lineTo(50, 270);
            potoCtx.moveTo(90, 120); potoCtx.lineTo(90, 250);
            // Capillary
            potoCtx.moveTo(50, 270); potoCtx.lineTo(550, 270);
            potoCtx.moveTo(90, 250); potoCtx.lineTo(140, 250); // Left of reservoir
            potoCtx.moveTo(180, 250); potoCtx.lineTo(550, 250); // Right of reservoir
            // Reservoir
            potoCtx.moveTo(140, 250); potoCtx.lineTo(140, 80); potoCtx.lineTo(120, 40);
            potoCtx.moveTo(180, 250); potoCtx.lineTo(180, 80); potoCtx.lineTo(200, 40);
            potoCtx.stroke();

            // Reservoir Tap (Stopcock)
            potoCtx.fillStyle = '#64748b';
            if (isResetting) {
                // Tap is Open (Vertical) allowing water to flow down
                potoCtx.fillRect(155, 190, 10, 30);
            } else {
                // Tap is Closed (Horizontal)
                potoCtx.fillRect(135, 200, 50, 10);
            }
            potoCtx.fillStyle = '#334155'; potoCtx.font = '12px Arial';
            potoCtx.fillText('Reservoir Tap', 125, 185);

            // Beaker Glass
            potoCtx.strokeStyle = 'rgba(148, 163, 184, 0.8)'; potoCtx.beginPath();
            potoCtx.moveTo(520, 190); potoCtx.lineTo(520, 320); potoCtx.lineTo(580, 320); potoCtx.lineTo(580, 190); potoCtx.stroke();

            // Rubber Bung
            potoCtx.fillStyle = '#475569'; potoCtx.fillRect(45, 130, 50, 25);

            // Plant
            potoCtx.strokeStyle = '#16a34a'; potoCtx.lineWidth = 8;
            potoCtx.beginPath(); potoCtx.moveTo(70, 160); potoCtx.lineTo(70, 110); potoCtx.quadraticCurveTo(70, 50, 50, 20); potoCtx.stroke();
            potoCtx.fillStyle = '#22c55e';
            potoCtx.beginPath(); potoCtx.ellipse(70, 120, 35, 12, Math.PI/6, 0, Math.PI*2); potoCtx.fill();
            potoCtx.beginPath(); potoCtx.ellipse(140, 90, 30, 12, -Math.PI/6, 0, Math.PI*2); potoCtx.fill();
            potoCtx.beginPath(); potoCtx.ellipse(50, 60, 30, 12, Math.PI/4, 0, Math.PI*2); potoCtx.fill();
            potoCtx.beginPath(); potoCtx.ellipse(120, 40, 25, 10, -Math.PI/4, 0, Math.PI*2); potoCtx.fill();

            // Particles
            vaporParticles.forEach(p => p.draw());
            usedParticles.forEach(p => p.draw());
        }

        function calculateCombinedRate() {
            let L = parseFloat(document.getElementById('slide-light').value);
            let T = parseFloat(document.getElementById('slide-temp').value);
            let W = parseFloat(document.getElementById('slide-wind').value);
            let H = parseFloat(document.getElementById('slide-humidity').value);

            document.getElementById('val-light').innerText = L + '%';
            document.getElementById('val-temp').innerText = T + '°C';
            document.getElementById('val-wind').innerText = W + ' km/h';
            document.getElementById('val-humidity').innerText = H + '%';

            // Multipliers normalized against baseline
            let rL = mathModels.light(L) / mathModels.light(50);
            let rT = mathModels.temp(T) / mathModels.temp(25);
            let rW = mathModels.wind(W) / mathModels.wind(10);
            let rH = mathModels.humidity(H) / mathModels.humidity(40);

            return Math.max(0.02, Math.min(rL * rT * rW * rH, 5.0));
        }

        function updatePotometerLogic() {
            // Handle Reservoir Reset Animation
            if (isResetting) {
                bubbleX += 5; // Fast movement back to start
                if (bubbleX >= startBubbleX) {
                    bubbleX = startBubbleX;
                    isResetting = false;
                    resetBtn.disabled = false;
                    resetBtn.innerText = "Open Reservoir Tap (Reset)";
                    // Resume play state if it was playing
                    if (playBtn.innerText === "Pause Simulation") isPlaying = true;
                }
            }

            if (!isPlaying || isResetting) return;

            let rateModifier = calculateCombinedRate();

            // Move bubble
            if (bubbleX > endBubbleX) {
                bubbleX -= 0.15 * rateModifier;
            } else {
                isPlaying = false;
                playBtn.innerText = "Play Simulation";
                bubbleX = endBubbleX; 
            }

            // Data Math
            totalUptake = (startBubbleX - bubbleX) * 0.15; 
            document.getElementById('stat-uptake').innerText = totalUptake.toFixed(1) + ' µL';
            document.getElementById('stat-transp').innerText = (totalUptake * 0.98).toFixed(1) + ' µL';
            document.getElementById('stat-used').innerText = (totalUptake * 0.02).toFixed(1) + ' µL';

            // Particle Spawning
            if (Math.random() < 0.3 * rateModifier && bubbleX > endBubbleX) {
                let leaf = leaves[Math.floor(Math.random() * leaves.length)];
                vaporParticles.push(new Particle(leaf.x, leaf.y, 'vapor'));
            }
            if (Math.random() < 0.02 * rateModifier && bubbleX > endBubbleX) {
                let leaf = leaves[Math.floor(Math.random() * leaves.length)];
                usedParticles.push(new Particle(leaf.x, leaf.y, 'used'));
            }

            for (let i = vaporParticles.length - 1; i >= 0; i--) {
                vaporParticles[i].update(rateModifier);
                if (vaporParticles[i].life > vaporParticles[i].maxLife) vaporParticles.splice(i, 1);
            }
            for (let i = usedParticles.length - 1; i >= 0; i--) {
                usedParticles[i].update();
                if (usedParticles[i].life > usedParticles[i].maxLife) usedParticles.splice(i, 1);
            }
        }

        function animationLoop() {
            updatePotometerLogic();
            drawPotometer();
            requestAnimationFrame(animationLoop);
        }

        // Attach listeners
        ['light', 'temp', 'wind', 'humidity'].forEach(factor => {
            document.getElementById(`slide-${factor}`).addEventListener('input', () => {
                if(activeFactor === factor) updateChart();
            });
        });

        playBtn.addEventListener('click', () => {
            if(bubbleX <= endBubbleX) { alert("Please reset the bubble using the reservoir tap first."); return; }
            isPlaying = !isPlaying;
            playBtn.innerText = isPlaying ? "Pause Simulation" : "Play Simulation";
        });

        resetBtn.addEventListener('click', () => {
            if (!isResetting && bubbleX < startBubbleX) {
                isResetting = true;
                isPlaying = false; // Pause standard logic while water flows
                resetBtn.disabled = true;
                resetBtn.innerText = "Flushing Water...";
                vaporParticles = []; usedParticles = [];
            }
        });

        // Init
        selectFactor('light');
        animationLoop();

    </script>
