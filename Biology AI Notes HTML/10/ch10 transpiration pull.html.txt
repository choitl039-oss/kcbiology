    <script>
        const canvas = document.getElementById('pullCanvas');
        const ctx = canvas.getContext('2d');

        let activeStep = 0; 
        let animationTimer = 0;
        let particles = [];

        // Setup 4 Mesophyll Cells (0 is next to xylem, 3 is next to air space)
        const cells = [
            { x: 260, y: 225, r: 50 },
            { x: 410, y: 225, r: 50 },
            { x: 560, y: 225, r: 50 },
            { x: 710, y: 225, r: 50 } // Touches air space
        ];
        const xylem = { x: 50, y: 50, w: 90, h: 350 };

        // Water Potential (WP) state for each cell (100 = High, 0 = Low)
        let currentWPs = [100, 100, 100, 100];
        let targetWPs =  [100, 100, 100, 100];
        
        // Booleans for particle flow rendering
        let flowEnabled = [false, false, false]; // [0->1, 1->2, 2->3]
        let xylemPullEnabled = false;
        let evaporationEnabled = false;

        const stepContent = {
            0: {
                title: "Overall Chain Reaction",
                text: "Observe the full chain reaction: evaporation at the edge lowers <span class='highlight-wp'>water potential</span>, pulling water from the next cell, creating a gradient that eventually triggers the <span class='highlight-pull'>Transpiration pull (蒸騰拉力)</span> from the xylem.",
                color: "#8b5cf6"
            },
            1: {
                title: "Step 1: Evaporation and WP Drop",
                text: "During transpiration, mesophyll cells near the air space lose water to the air space. Their <span class='highlight-wp'>water potential lowers</span>.",
                color: "#3b82f6"
            },
            2: {
                title: "Step 2: Osmosis Chain Reaction",
                text: "Water moves from neighbouring cells into these cells by <span class='highlight-osmosis'>osmosis</span> (due to the water potential difference). This is repeated across the layer of mesophyll cells, creating a gradient.",
                color: "#3b82f6"
            },
            3: {
                title: "Step 3: Transpiration Pull",
                text: "Eventually, water moves out of the xylem vessels by osmosis to replace the water loss in mesophyll cells. <span class='highlight-pull'>Transpiration pull (蒸騰拉力)</span> is created to pull water up the xylem vessels from roots.",
                color: "#059669"
            }
        };

        class FlowParticle {
            constructor(type, startX, startY, endX, endY) {
                this.x = startX;
                this.y = startY;
                this.endX = endX;
                this.endY = endY;
                this.type = type; // 'osmosis', 'evap', 'pull'
                this.life = 0;
                this.maxLife = 40 + Math.random() * 20;
            }
            update() {
                this.life++;
                if (this.type === 'pull') {
                    // Shoot straight up the xylem fast
                    this.y -= 6;
                    this.x += (Math.random() - 0.5) * 2;
                } else if (this.type === 'osmosis') {
                    // Move smoothly from one cell to the next
                    this.x += (this.endX - this.x) * 0.05;
                    this.y += (this.endY - this.y) * 0.05 + (Math.random()-0.5)*2;
                } else if (this.type === 'evap') {
                    // Float up and right into air space
                    this.x += 2 + Math.random() * 2;
                    this.y -= 1 + Math.random() * 2;
                }
            }
            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.type === 'pull' ? 4 : 3, 0, Math.PI * 2);
                ctx.fillStyle = '#eab308'; // Orange/Yellow water particles so they contrast with blue WP
                ctx.fill();
            }
        }

        function spawnParticles() {
            // Evaporation (Cell 3 to Air)
            if (evaporationEnabled && Math.random() < 0.3) {
                particles.push(new FlowParticle('evap', cells[3].x + 40, cells[3].y + (Math.random()-0.5)*40, 0, 0));
            }
            // Osmosis Chain Reaction
            for (let i = 0; i < 3; i++) {
                if (flowEnabled[i] && Math.random() < 0.2) {
                    particles.push(new FlowParticle('osmosis', cells[i].x, cells[i].y, cells[i+1].x, cells[i+1].y));
                }
            }
            // Xylem to Cell 0
            if (xylemPullEnabled && Math.random() < 0.3) {
                particles.push(new FlowParticle('osmosis', xylem.x + xylem.w, cells[0].y, cells[0].x, cells[0].y));
            }
            // Transpiration Pull (Up the Xylem)
            if (xylemPullEnabled && Math.random() < 0.8) { // Very fast spawn rate
                particles.push(new FlowParticle('pull', xylem.x + Math.random()*xylem.w, xylem.y + xylem.h, 0, 0));
            }
        }

        function updateSimulationLogic() {
            animationTimer++;
            
            // Smoothly transition current WP to target WP
            for(let i=0; i<4; i++) {
                currentWPs[i] += (targetWPs[i] - currentWPs[i]) * 0.03;
            }

            // Logic State Machine based on selected Step
            if (activeStep === 1) {
                // Step 1: Only cell 3 loses water. No flow from neighbors.
                targetWPs = [100, 100, 100, 30];
                flowEnabled = [false, false, false];
                xylemPullEnabled = false;
                evaporationEnabled = true;
            } 
            else if (activeStep === 2) {
                // Step 2: Chain reaction propagates from right to left based on timer
                targetWPs = [100, 100, 100, 100];
                evaporationEnabled = true;
                
                if (animationTimer > 20) targetWPs[3] = 30; // Cell 3 drops
                if (animationTimer > 70) { 
                    targetWPs[2] = 50;  // Cell 2 drops
                    flowEnabled[2] = true; // Flow 2->3 starts
                }
                if (animationTimer > 140) { 
                    targetWPs[1] = 70; // Cell 1 drops
                    flowEnabled[1] = true; // Flow 1->2 starts
                }
                // Xylem remains disconnected to highlight just the mesophyll chain
                flowEnabled[0] = false;
                xylemPullEnabled = false;
            } 
            else if (activeStep === 3) {
                // Step 3: Xylem connects. Gradient is stable. Pull activates.
                targetWPs = [85, 70, 55, 40];
                flowEnabled = [true, true, true];
                xylemPullEnabled = true;
                evaporationEnabled = true;
            }
            else if (activeStep === 0) {
                // Overall: Show the whole sequence from 0 to xylem pull
                if (animationTimer < 20) {
                    targetWPs = [100, 100, 100, 100];
                    flowEnabled = [false, false, false];
                    xylemPullEnabled = false;
                    evaporationEnabled = false;
                }
                if (animationTimer > 20) { evaporationEnabled = true; targetWPs[3] = 40; }
                if (animationTimer > 80) { flowEnabled[2] = true; targetWPs[2] = 55; }
                if (animationTimer > 140) { flowEnabled[1] = true; targetWPs[1] = 70; }
                if (animationTimer > 200) { 
                    flowEnabled[0] = true; 
                    xylemPullEnabled = true; 
                    targetWPs[0] = 85; 
                }
            }

            // Update particles
            spawnParticles();
            for (let i = particles.length - 1; i >= 0; i--) {
                particles[i].update();
                if (particles[i].life > particles[i].maxLife || particles[i].y < 50 || particles[i].x > 850) {
                    particles.splice(i, 1);
                }
            }
        }

        function drawGauge(x, y, wpValue, label) {
            // Draw background bar
            ctx.fillStyle = '#e2e8f0';
            ctx.fillRect(x - 30, y, 60, 12);
            
            // Draw fill level based on WP
            // Color shifts from light blue (low) to deep blue (high)
            const fillWidth = Math.max(0, (wpValue / 100) * 60);
            const intensity = Math.floor(255 - (wpValue / 100) * 100);
            ctx.fillStyle = `rgb(37, 99, ${255 - intensity/2})`; 
            ctx.fillRect(x - 30, y, fillWidth, 12);
            
            // Border
            ctx.strokeStyle = '#94a3b8';
            ctx.lineWidth = 1;
            ctx.strokeRect(x - 30, y, 60, 12);

            // Text
            ctx.fillStyle = '#475569';
            ctx.font = '12px bold Segoe UI';
            ctx.textAlign = 'center';
            ctx.fillText(label + ` Ψ: ${Math.round(wpValue)}%`, x, y + 25);
        }

        function drawArrow(x1, x2, y, label) {
            ctx.strokeStyle = '#2563eb';
            ctx.lineWidth = 3;
            ctx.beginPath();
            ctx.moveTo(x1, y);
            ctx.lineTo(x2, y);
            ctx.stroke();
            
            // Arrowhead
            ctx.beginPath();
            ctx.moveTo(x2, y);
            ctx.lineTo(x2 - 8, y - 6);
            ctx.lineTo(x2 - 8, y + 6);
            ctx.fillStyle = '#2563eb';
            ctx.fill();

            // Label
            ctx.fillStyle = '#1d4ed8';
            ctx.font = '12px bold Arial';
            ctx.fillText(label, (x1+x2)/2, y - 10);
        }

        function drawScene() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Draw Xylem Pipe
            ctx.fillStyle = '#cbd5e1';
            ctx.fillRect(xylem.x, xylem.y, xylem.w, xylem.h);
            ctx.strokeStyle = '#94a3b8';
            ctx.lineWidth = 2;
            ctx.strokeRect(xylem.x, xylem.y, xylem.w, xylem.h);
            
            ctx.fillStyle = '#1e293b';
            ctx.font = '16px bold Segoe UI';
            ctx.textAlign = 'center';
            ctx.fillText('XYLEM', xylem.x + xylem.w/2, xylem.y - 15);
            ctx.font = '12px Segoe UI';
            ctx.fillText('(High Ψ)', xylem.x + xylem.w/2, xylem.y + 30);

            // Transpiration Pull Label (appears in step 3)
            if (xylemPullEnabled) {
                ctx.fillStyle = '#059669';
                ctx.font = 'bold 16px Arial';
                ctx.fillText('↑ Transpiration Pull', xylem.x + xylem.w/2, xylem.y + 150);
                ctx.fillText('↑ 蒸騰拉力', xylem.x + xylem.w/2, xylem.y + 175);
            }

            // Draw Mesophyll Cells
            for (let i = 0; i < cells.length; i++) {
                const c = cells[i];
                const wp = currentWPs[i];

                // Calculate color based on WP (100 = rich blue, 0 = very pale blue)
                const blueValue = Math.max(150, 255 - wp);
                ctx.fillStyle = `rgba(37, 99, 235, ${wp / 100 * 0.7 + 0.1})`; // dynamic opacity interior
                
                ctx.beginPath();
                ctx.arc(c.x, c.y, c.r, 0, Math.PI * 2);
                ctx.fill();
                
                ctx.strokeStyle = '#16a34a'; // Green cell wall
                ctx.lineWidth = 4;
                ctx.stroke();

                // Draw WP Gauge below cell
                drawGauge(c.x, c.y + c.r + 15, wp, `Cell ${i+1}`);
            }

            // Draw Osmosis Difference Arrows between cells
            // Only draw if there is a noticeable WP gradient driving it
            for (let i = 0; i < 3; i++) {
                if (currentWPs[i] - currentWPs[i+1] > 10) {
                    drawArrow(cells[i].x + cells[i].r + 5, cells[i+1].x - cells[i+1].r - 5, cells[i].y - 30, 'ΔΨ (Osmosis)');
                }
            }

            // Draw Xylem to Cell 0 Arrow
            if (xylemPullEnabled && (100 - currentWPs[0] > 5)) {
                drawArrow(xylem.x + xylem.w + 5, cells[0].x - cells[0].r - 5, cells[0].y - 30, 'Osmosis');
            }

            // Draw Air Space Label
            ctx.fillStyle = '#64748b';
            ctx.font = '18px bold Segoe UI';
            ctx.fillText('Air Space', 800, 225);

            // Draw Particles
            particles.forEach(p => p.draw());
        }

        function animate() {
            updateSimulationLogic();
            drawScene();
            requestAnimationFrame(animate);
        }

        window.setStep = function(stepNum) {
            activeStep = stepNum;
            animationTimer = 0; // Reset timer for chain reactions
            
            // UI Update Buttons
            document.querySelectorAll('.step-btn').forEach(btn => btn.classList.remove('active'));
            document.getElementById('btn-' + stepNum).classList.add('active');

            // UI Update Explanation Text
            const info = stepContent[stepNum];
            const explanationBox = document.getElementById('explanation-box');
            
            explanationBox.innerHTML = `<h3>${info.title}</h3><p>${info.text}</p>`;
            explanationBox.style.borderLeftColor = info.color;

            // Instantly reset WP values if switching away from continuous
            if (stepNum !== 0 && stepNum !== activeStep) {
                currentWPs = [100, 100, 100, 100];
            }
            
            // Clear existing particles to keep screen clean on step switch
            particles = [];
        }

        // Start animation
        setStep(0);
        animate();
    </script>
